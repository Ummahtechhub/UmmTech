rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    function isSignedIn() {
      return request.auth != null;
    }

    function isSmallEnough() {
      return request.resource == null || request.resource.size < 20 * 1024 * 1024;
    }

    match /{allPaths=**} {
      allow read: if isSignedIn();
      allow write: if isSignedIn() && isSmallEnough();
    }
  }
}
